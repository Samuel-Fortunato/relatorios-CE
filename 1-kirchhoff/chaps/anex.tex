\subsection{Equações utilizadas:}

\paragraph{Erro relativo}
\begin{equation}
    Er(\%) =
    \frac{\left|x_{\text{experimental}} - x_{\text{teórico}}\right|}
    {x_{\text{teórico}}}
    \times 100\%
\end{equation}

\paragraph{Desvio experimental}
\begin{equation}
    D = x_{\text{experimental}} - x_{\text{teórico}}
\end{equation}

\paragraph{Lei de Ohm}
\begin{equation}
    V = RI \iff R = \frac{V}{I} \iff I = \frac{V}{R}
\end{equation}

\paragraph{Lei dos nós}
\begin{equation*}
    \sum_{i=1}^{n} I_i = 0
\end{equation*}

\paragraph{Lei das malhas}
\begin{equation*}
    \sum_{i=1}^{n} V_i = 0
\end{equation*}
